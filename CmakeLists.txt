cmake_minimum_required(VERSION 3.6)

# Name of the project (will be the name of the plugin)
project(measurement-kit)

# Essential include files to build a node addon,
# you should add this line in every CMake.js based project.
include_directories(${CMAKE_JS_INC})

# Declare the location of the source files
file(
    GLOB
    SOURCE_FILES
    "src/*.cpp"
    "src/*/*.cpp"
    "src/*.hpp"
    "src/*/*.hpp"
)

# Build MeasurementKit as part of building the bindings
set(MEASUREMENTKIT_ROOT "${CMAKE_SOURCE_DIR}/deps/measurement-kit/")
include("${MEASUREMENTKIT_ROOT}/cmake/Common.cmake")
include("${MEASUREMENTKIT_ROOT}/cmake/LibMeasurementKit.cmake")
set(MEASUREMENTKIT_LIBRARIES "measurement_kit_static" "${MK_LIBS}")

# This line will tell CMake that we're building a shared library
# from the above source files
# named after the project's name
add_library(${PROJECT_NAME} SHARED ${SOURCE_FILES})

# Gives our library file a .node extension without any "lib" prefix
set_target_properties(${PROJECT_NAME} PROPERTIES PREFIX "" SUFFIX ".node")

message(STATUS "LIBRARIES ${MEASUREMENTKIT_LIBRARIES} | ${MEASUREMENTKIT_INCLUDE_DIRS}")
# Link everything together
target_link_libraries(${PROJECT_NAME}
    ${MEASUREMENTKIT_LIBRARIES};${CMAKE_JS_LIB})
